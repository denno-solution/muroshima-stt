# Dockerfile
FROM node:18-slim

# 必要なパッケージインストール（ffmpeg含む）
RUN apt-get update && apt-get install -y ffmpeg && \
    apt-get clean && rm -rf /var/lib/apt/lists/*

# 作業ディレクトリ
WORKDIR /app

# package.jsonとpackage-lock.jsonをコピーして依存解決
COPY package*.json ./
RUN npm install

# アプリのコードをコピー
COPY . .

# Cloud Run用ポート環境変数
ENV PORT=8080

# アプリ起動
CMD ["node", "index.js"]
